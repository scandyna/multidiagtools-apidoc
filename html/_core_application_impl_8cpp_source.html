49 void CoreApplicationImpl::enableFileLogging(CoreApplicationImpl::LogFileNameFormat format)
50 {
51  using Mdt::ErrorLogger::Logger;
52  using Mdt::ErrorLogger::FileBackend;
53 
54  if(isFileLoggingEnabled()){
55  return;
56  }
57  // Create log directory
58  QDir logDirectory( logDirectoryPath() );
59  if(!logDirectory.exists()){
60  if(!logDirectory.mkpath( logDirectory.absolutePath() )){
61  qWarning() << tr("Mdt::CoreApplication: could not create log directory '%1'. File logging will be disabled.").arg(logDirectory.absolutePath());
62  return;
63  }
64  }
65  Q_ASSERT(logDirectory.exists());
66  // Build log file path
67  QString logFileName;
68  switch(format){
69  case ApplicationName:
70  logFileName = QCoreApplication::applicationName();
71  break;
72  case ApplicationNameAndPid:
73  logFileName = QCoreApplication::applicationName() % QLatin1Char('_') % QString::number( QCoreApplication::applicationPid() );
74  break;
75  }
76  mLogFilePath = QDir::cleanPath( logDirectory.absolutePath() % QLatin1Char('/') % logFileName % QLatin1String(".log") );
77  // Add file backend to the error logger
78  auto *logFileBackend = Logger::addBackend<FileBackend>(Logger::ExecuteInSeparateThread);
79  Q_ASSERT(logFileBackend != nullptr);
80  if(!logFileBackend->setLogFilePath(mLogFilePath)){
81  qWarning() << tr("Mdt::CoreApplication: log file '%1' could not be open, no log will come to it.").arg(mLogFilePath);
82  return;
83  }
84 }
