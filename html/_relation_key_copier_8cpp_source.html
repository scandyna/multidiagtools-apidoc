72 bool RelationKeyCopier::copyAllKeyData(RowRange childModelRowRange, const QModelIndex & parent)
73 {
74  if(mParentModel.isNull() || mChildModel.isNull()){
75  return false;
76  }
77  if(!childModelRowRange.isValid()){
78  return false;
79  }
80  if(childModelRowRange.rowCount() > mChildModel->rowCount(parent)){
81  return false;
82  }
83  if( (mParentModelCurrentRow < 0) || (mParentModelCurrentRow >= mParentModel->rowCount()) ){
84  return false;
85  }
86  const int last = childModelRowRange.lastRow();
87  for(int row = childModelRowRange.firstRow(); row <= last; ++row){
88  if( !copyKeyDataForRow(row) ){
89  return false;
90  }
91  }
92  return true;
93 }
