301 QVariant FormatProxyModel::data(const QModelIndex & index, int role) const
302 {
303  if(!index.isValid()){
304  return QVariant();
305  }
306  if( (index.row() >= rowCount()) || (index.column() >= columnCount()) ){
307  return QVariant();
308  }
309  switch(role){
310  case Qt::TextAlignmentRole:
311  return textAlignment(index.row(), index.column());
312  case Qt::FontRole:
313  return textFont(index.row(), index.column());
314  case Qt::ForegroundRole:
315  return foregroundBrush(index.row(), index.column());
316  case Qt::BackgroundRole:
317  return backgroundBrush(index.row(), index.column());
318  default:
319  ;
320  }
321  return QIdentityProxyModel::data(index, role);
322 }

324 void FormatProxyModel::signalFormatChangedForIndex(int row, int column, int role)
325 {
326  Q_ASSERT(row >= 0);
327  Q_ASSERT(column >= 0);
328 
329  if(sourceModel() == nullptr){
330  return;
331  }
332  Q_ASSERT(row < rowCount());
333  Q_ASSERT(column < columnCount());
334 
335  const auto idx = index(row, column);
336  emit dataChanged(idx, idx, {role});
337 }

339 void FormatProxyModel::signalFormatChangedForRow(int row, int role)
340 {
341  Q_ASSERT(row >= 0);
342 
343  if(sourceModel() == nullptr){
344  return;
345  }
346  Q_ASSERT(row < rowCount());
347 
348  const auto topLeft = index(row, 0);
349  const auto bottomRight = index(row, columnCount()-1);
350  emit dataChanged(topLeft, bottomRight, {role});
351 }

353 void FormatProxyModel::signalFormatChangedForColumn(int column, int role)
354 {
355  Q_ASSERT(column >= 0);
356 
357  if(sourceModel() == nullptr){
358  return;
359  }
360  Q_ASSERT(column < columnCount());
361 
362  const int n = rowCount();
363  for(int row = 0; row < n; ++row){
364  const auto idx = index(row, column);
365  emit dataChanged(idx, idx, {role});
366  }
367 }
