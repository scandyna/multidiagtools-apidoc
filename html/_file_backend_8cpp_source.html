77 void FileBackend::logError(const Error & error)
78 {
79  QFile file(mFilePath);
80 
81  // Open log file
82  if(!openFile(file)){
83  return;
84  }
85  // backup log file if needed
86  if(file.size() >= mMaxFileSize){
87  file.close();
88  backupLogFile();
89  if(!openFile(file)){
90  return;
91  }
92  }
93  Q_ASSERT(file.isOpen());
94  Q_ASSERT(file.isWritable());
95  // Write error to file
96  QTextStream stream(&file);
97  stream << formatError(error) << "\n";
98  stream.flush();
99  file.close();
100 }

102 void FileBackend::cleanup()
103 {
104  if(mFilePath.isEmpty()){
105  return;
106  }
107  // If we have a empty log file, we remove it
108  QFile file(mFilePath);
109  if(!file.open(QFile::ReadOnly | QFile::Text)){
110  qWarning() << tr("Mdt::ErrorLogger::FileBackend::cleanup() : could not open file %1.\n Error: %2")
111  .arg(mFilePath, file.errorString());
112  return;
113  }
114  if(file.size() > 0){
115  file.close();
116  return;
117  }
118  if(!file.remove()){
119  qWarning() << tr("Mdt::ErrorLogger::FileBackend::cleanup() : could not remove file %1.\n Error: %2")
120  .arg(mFilePath, file.errorString());
121  }
122 }
