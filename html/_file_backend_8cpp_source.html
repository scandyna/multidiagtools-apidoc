74 void FileBackend::logError(const Error & error)
75 {
76  QFile file(pvFilePath);
77 
78  // Open log file
79  if(!openFile(file)){
80  return;
81  }
82  // backup log file if needed
83  if(file.size() >= pvMaxFileSize){
84  file.close();
85  backupLogFile();
86  if(!openFile(file)){
87  return;
88  }
89  }
90  Q_ASSERT(file.isOpen());
91  Q_ASSERT(file.isWritable());
92  // Write error to file
93  QTextStream stream(&file);
94  stream << getErrorStackString(error) << getErrorString(error) << QStringLiteral("\n");
95  stream.flush();
96  file.close();
97 }

112 QString FileBackend::getErrorString(const Error & error) const
113 {
114  QString str;
115  QString informativeText;
116  QString lineBegin;
117 
118  // Prepare line beginning
119  lineBegin = QDateTime::currentDateTime().toString(QLatin1String("yyyy.MM.dd hh:mm:ss")) + QStringLiteral(" ") \
120  % errorLevelText(error.level()) % QStringLiteral(": ");
121  // Build message
122  str = lineBegin % QStringLiteral("In ") % error.functionName() % QStringLiteral("\n") \
123  % lineBegin + QStringLiteral("-> ") % tr("Text: ") % error.text() % QStringLiteral("\n");
124  informativeText = error.informativeText();
125  if(!informativeText.isEmpty()){
126  str += lineBegin % QStringLiteral("-> ") % error.informativeText() % QStringLiteral("\n");
127  }
128  str += lineBegin % QStringLiteral("-> ") % tr("File: ") % error.fileName() % QStringLiteral("\n") \
129  % lineBegin % QStringLiteral("-> ") % tr("Line: ") % QString::number(error.fileLine());
130 
131  return str;
132 }
