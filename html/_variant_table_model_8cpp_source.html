135 bool VariantTableModel::setData(const QModelIndex& index, const QVariant& value, int role)
136 {
137  if(!index.isValid()){
138  return false;
139  }
140  Q_ASSERT(index.row() >= 0);
141  Q_ASSERT(index.column() >= 0);
142  if( (index.row() >= (int)mData.size()) || (index.column() >= mColumnCount) ){
143  return false;
144  }
145  if( (role == Qt::DisplayRole) || (role == Qt::EditRole) ){
146  mData[index.row()].setData(index.column(), value, role);
147  if(mPassRolesInDataChanged){
148  if(mStorageRule == VariantTableModelStorageRule::SeparateDisplayAndEditRoleData){
149  emit dataChanged(index, index, {role});
150  }else{
151  emit dataChanged(index, index, {Qt::DisplayRole, Qt::EditRole});
152  }
153  }else{
154  emit dataChanged(index, index);
155  }
156  return true;
157  }
158  return false;
159 }
