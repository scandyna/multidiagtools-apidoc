81  searchPathList.setPathPrefixList(pathPrefixList);

86  for(const auto & path : pathList){
87  qDebug() << " Searchin in " << path;

109 bool isTranslationDirectory(const QDir& dir, const QString& expectedQmFileBaseName)
110 {
111  if(!dir.exists()){
112  return false;
113  }
114  const auto fiList = dir.entryInfoList(QDir::Files);
115  for(const auto & fi : fiList){
116  if(expectedQmFileBaseName.isEmpty()){
117  if(QString::compare(fi.suffix(), QLatin1String("qm"), Qt::CaseInsensitive) == 0){
118  return true;
119  }
120  }else{
121  QmFileName qmFile(fi.fileName());
122  if(qmFile.baseName() == expectedQmFileBaseName){
123  return true;
124  }
125  }
126  }
127  return false;
128 }
