45 bool FileInputIteratorSharedData::setSource(QIODevice* device, const QByteArray& encoding)
46 {
47  Q_ASSERT(device != nullptr);
48 
49  // Delete possibly allocated decoder (it is a nullptr or a valid pointer)
50  delete mDecoder;
51  mDecoder = nullptr;
52  // Set possibly referenced divice to nullptr
53  mDevice = nullptr;
54  // Clear buffer
55  clearUnicodeBuffer();
56  /*
57  * Find a codec for requested encoding
58  * Note: we not have to manage codec lifetime,
59  * see QTextCodec::~QTextCodec()
60  */
61  auto *codec = QTextCodec::codecForName(encoding);
62  if(codec == nullptr){
63  const QString msg = tr("Could not find a codec for requested encoding '%1'").arg( QString::fromLatin1(encoding) );
64  mLastError = mdtErrorNew(msg, Mdt::Error::Critical, "FileInputIteratorSharedData");
65  mLastError.commit();
66  return false;
67  }
68  // Allocate our decoder
69  Q_ASSERT(codec != nullptr);
70  mDecoder = codec->makeDecoder();
71  /*
72  * Check that device is open
73  * We must do it at runtime,
74  * because it can be closed at any time
75  * (f.ex. a file on a removable device)
76  */
77  if(!device->isOpen()){
78  QString msg = tr("Device is not open.");
79  mLastError = mdtErrorNew(msg, Mdt::Error::Critical, "FileInputIteratorSharedData");
80  msg = device->errorString();
81  if(!msg.isEmpty()){
82  const auto error = mdtErrorNew(msg, Mdt::Error::Critical, "FileInputIteratorSharedData");
83  mLastError.stackError(error);
84  }
85  mLastError.commit();
86  return false;
87  }
88  Q_ASSERT(device->isReadable());
89  mDevice = device;
90  /*
91  * If device is allready at end, its not a error,
92  * but we do not try to read from it.
93  */
94  if(device->atEnd()){
95  return true;
96  }
97  /*
98  * Try to read and decode the first chunck of data from device
99  */
100  while(mCurrentPos == mEnd){
101  if(!readMore()){
102  return false;
103  }
104  }
105  Q_ASSERT(mCurrentPos != mEnd);
106 
107  return true;
108 }

110 bool FileInputIteratorSharedData::advance()
111 {
112  Q_ASSERT(mDevice);
113  Q_ASSERT(mDecoder);
114 
115  // Check if we have cached data
116  if(mCurrentPos != mEnd){
117  Q_ASSERT(mCurrentPos < mEnd);
118  ++mCurrentPos;
119  }
120  // Check if we have to read more data
121  while(mCurrentPos == mEnd){
122  if(!readMore()){
123  Q_ASSERT(mCurrentPos == mEnd);
124  return false;
125  }
126  if(mDevice->atEnd()){
127  return true;
128  }
129  }
130 
131  return true;
132 }

134 bool FileInputIteratorSharedData::readMore()
135 {
136  Q_ASSERT(mDevice);
137  Q_ASSERT(mDecoder);
138  Q_ASSERT(mRawDataBuffer.capacity() > 0);
139 
140  // Clear unicode buffer (also enshure that iterators are valid)
141  clearUnicodeBuffer();
142  /*
143  * Check that device is open
144  * We must do it at runtime,
145  * because it can be closed at any time
146  * (f.ex. a file on a removable device)
147  */
148  if(!mDevice->isOpen()){
149  QString msg = tr("Device was closed.");
150  mLastError = mdtErrorNew(msg, Mdt::Error::Critical, "FileInputIteratorSharedData");
151  msg = mDevice->errorString();
152  if(!msg.isEmpty()){
153  auto error = mdtErrorNew(msg, Mdt::Error::Critical, "FileInputIteratorSharedData");
154  mLastError.stackError(error);
155  }
156  mLastError.commit();
157  return false;
158  }
159  Q_ASSERT(mDevice->isReadable());
160  // Read a chunk of data from device
161  auto n = mDevice->read(mRawDataBuffer.data(), mRawDataBuffer.capacity());
162  if(n < 0){
163  QString msg = tr("Read from device failed.");
164  mLastError = mdtErrorNew(msg, Mdt::Error::Critical, "FileInputIteratorSharedData");
165  msg = mDevice->errorString();
166  if(!msg.isEmpty()){
167  auto error = mdtErrorNew(msg, Mdt::Error::Critical, "FileInputIteratorSharedData");
168  mLastError.stackError(error);
169  }
170  mLastError.commit();
171  return false;
172  }
173  // Decode readen data into unicode buffer and update iterators
174  mDecoder->toUnicode(&mUnicodeBuffer, mRawDataBuffer.data(), n);
175  mCurrentPos = mUnicodeBuffer.cbegin();
176  mEnd = mUnicodeBuffer.cend();
177 
178  return true;
179 }
