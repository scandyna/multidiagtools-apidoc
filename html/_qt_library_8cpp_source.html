36 LibraryInfoList QtLibrary::findLibraryPlugins(const LibraryInfo & qtLibrary, const PathList & searchFirstPathPrefixList)
37 {
38  LibraryInfoList plugins;
39 
40  PathList pathPrefixList = searchFirstPathPrefixList;
41  pathPrefixList.appendPathList( PathList::getSystemLibraryPathList() );
42  SearchPathList searchPathList;
43  searchPathList.setPathPrefixList(pathPrefixList);
44  searchPathList.setPathSuffixList({"qt5/plugins","plugins"});
45 
46  const auto pluginDirectories = getPluginsDirectories( module(qtLibrary) );
47  const auto pathList = searchPathList.pathList();
48  const auto os = LibraryName::operatingSystem(qtLibrary.libraryName().fullName());
49  Q_ASSERT(os != OperatingSystem::Unknown);
50 
51  Console::info(2) << " searching plugins for library " << qtLibrary.libraryName().name();
52  for(const auto & path : pathList){
53  Console::info(3) << " searching in " << path << " ...";
54  plugins = findPluginsInDirectories(path, pluginDirectories, os);
55  if(!plugins.isEmpty()){
56  break;
57  }
58  }
59 
60  return plugins;
61 }

134 QStringList QtLibrary::getPluginsDirectories(QtModule module)
135 {
136  QStringList directories;
137 
138  /*
139  * Note about Qt Platform Abstraction:
140  * QPA is not Qt module that produces a binary library.
141  * By seeing dependencies, we deduced to which module to attach each QPA directory
142  */
143  switch(module){
144  case QtModule::Unknown:
145  break;
146  case QtModule::Core:
147  // QPA: Qt Core seems appropriate for platforms
148  directories = QStringList{"platforms"};
149  break;
150  case QtModule::Gui:
151  // QPA: Qt GUI seems appropriate for platforminputcontexts and platformthemes
152  directories = QStringList{"accessiblebridge","imageformats","pictureformats","platforminputcontexts","platformthemes"};
153  break;
154  case QtModule::Multimedia:
155  directories = QStringList{"audio","video/declarativevideobackend","video/bufferpool","playlistformats","resourcepolicy","mediaservice","video/videonode"};
156  break;
157  case QtModule::MultimediaWidgets:
158  break;
159  case QtModule::Network:
160  directories = QStringList{"bearer"};
161  break;
162  case QtModule::Positioning:
163  directories = QStringList{"position"};
164  break;
165  case QtModule::PrintSupport:
166  directories = QStringList{"printsupport"};
167  break;
168  case QtModule::Quick:
169  directories = QStringList{"scenegraph"};
170  break;
171  case QtModule::Sensors:
172  directories = QStringList{"sensorgestures","sensors"};
173  break;
174  case QtModule::Sql:
175  directories = QStringList{"sqldrivers"};
176  break;
177  case QtModule::Svg:
178  directories = QStringList{"iconengines"};
179  break;
180  case QtModule::Widgets:
181  directories = QStringList{"accessible","styles"};
182  break;
183  case QtModule::Qml:
184  case QtModule::Test:
185  case QtModule::DBus:
186  break;
187  }
188 
189  return directories;
190 }

255 LibraryInfoList QtLibrary::findPluginsInDirectories(const QString & pathPrefix, const QStringList& directories, OperatingSystem os)
256 {
257  LibraryInfoList plugins;
258 
259  for(const auto & directory : directories){
260  QDir dir( QDir::cleanPath(pathPrefix + "/" + directory) );
261  if(dir.exists()){
262  const auto fiList = dir.entryInfoList(QDir::Files);
263  for(const auto & fi : fiList){
264  // When do cross compilation, don't bring wrong libraries to the list
265  if(LibraryName::operatingSystem(fi.fileName()) == os){
266  Console::info(4) << " found " << fi.fileName();
267  LibraryInfo li;
268  li.setLibraryPlatformName(fi.fileName());
269  li.setAbsoluteFilePath(fi.absoluteFilePath());
270  plugins.addLibrary(li);
271  }
272  }
273  }
274  }
275 
276  return plugins;
277 }
