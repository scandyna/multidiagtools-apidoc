122 QStringList QtLibrary::getPluginsDirectories(QtModule module)
123 {
124  QStringList directories;
125 
126  /*
127  * Note about Qt Platform Abstraction:
128  * QPA is not Qt module that produces a binary library.
129  * By seeing dependencies, we deduced to which module to attach each QPA directory
130  */
131  switch(module){
132  case QtModule::Unknown:
133  break;
134  case QtModule::Core:
135  // QPA: Qt Core seems appropriate for platforms
136  directories = QStringList{"platforms"};
137  break;
138  case QtModule::Gui:
139  // QPA: Qt GUI seems appropriate for platforminputcontexts and platformthemes
140  directories = QStringList{"accessiblebridge","imageformats","pictureformats","platforminputcontexts","platformthemes"};
141  break;
142  case QtModule::Multimedia:
143  directories = QStringList{"audio","video/declarativevideobackend","video/bufferpool","playlistformats","resourcepolicy","mediaservice","video/videonode"};
144  break;
145  case QtModule::MultimediaWidgets:
146  break;
147  case QtModule::Network:
148  directories = QStringList{"bearer"};

151  directories = QStringList{"position"};
152  break;
153  case QtModule::PrintSupport:
154  directories = QStringList{"printsupport"};
155  break;
156  case QtModule::Quick:
157  directories = QStringList{"scenegraph"};
158  break;
159  case QtModule::Sensors:
160  directories = QStringList{"sensorgestures","sensors"};
161  break;
162  case QtModule::Sql:
163  directories = QStringList{"sqldrivers"};
164  break;
165  case QtModule::Svg:
166  directories = QStringList{"iconengines"};

191 QString QtLibrary::findPluginsRoot(const PathList & pathPrefixList)
192 {
193  QString pluginsRoot;
194  SearchPathList searchPathList;
195  searchPathList.setIncludePathPrefixes(true);
196  if(pathPrefixList.isEmpty()){
197  searchPathList.setPathPrefixList( PathList::getSystemLibraryPathList() );
198  }else{
199  searchPathList.setPathPrefixList(pathPrefixList);
200  }
201  searchPathList.setPathSuffixList({"qt5",".."});
202  const auto pathList = searchPathList.pathList();
203 
204  for(const auto & path : pathList){
205  qDebug() << "Searchin in " << path;
206  QDir dir( QDir::cleanPath(path + "/plugins") );
207  if(dir.exists()){
208  pluginsRoot = dir.absolutePath();
209  qDebug() << " - found: " << pluginsRoot;
210  return pluginsRoot;
211  }
212  }
213 
214  return pluginsRoot;
215 }

279 QtPluginInfoList QtLibrary::findPluginsInDirectories(const QString & pluginsRoot, const QStringList& directories, const LibraryInfo & qtLibrary)
280 {
281  QtPluginInfoList plugins;
282  const auto qtLibraryName = qtLibrary.libraryName();
283 
284  for(const auto & directory : directories){
285  QDir dir( QDir::cleanPath(pluginsRoot + "/" + directory) );
286  if(dir.exists()){
287  const auto fiList = dir.entryInfoList(QDir::Files);
288  for(const auto & fi : fiList){
289  // Create LibraryName and compare extension with given qtLibrary
290  LibraryName plugin(fi.fileName());
291  if( compareStringsCi(plugin.extension(), qtLibraryName.extension()) && (plugin.hasNameDebugSuffix() == qtLibraryName.hasNameDebugSuffix()) ){
292  Console::info(4) << " found " << fi.fileName();
293  QtPluginInfo qpi;
294  qpi.setLibraryPlatformName(fi.fileName());
295  qpi.setAbsoluteFilePath(fi.absoluteFilePath());
296  qpi.setDirectoryName(directory);
297  plugins.addPlugin(qpi);
298  }
299  }
300  }
301  }
302 
303  return plugins;
304 }
