61 void LikeExpressionRegexTransform::escapeBackslash(QString & str)
62 {
63  if(str.isEmpty()){
64  return;
65  }
66  int i = str.indexOf('\\');
67  while( (i > -1) && (i < str.size()) ){
68  /*
69  * If we matched '\' at last char, we never have a wildcard after -> escape.
70  * Else, we have to check if following char is a wildcard
71  */
72  if( i == (str.size()-1) ){
73  str.insert(i, '\\');
74  }else{
75  Q_ASSERT( i < (str.size()-1) );
76  if( !isTokenWildcard(str.at(i+1)) ){
77  str.insert(i, '\\');
78  }
79  }
80  /*
81  * Increment and find next match
82  */
83  i += 2;
84  if(i < str.size()){
85  i = str.indexOf('\\', i);
86  }
87  }
88 }
