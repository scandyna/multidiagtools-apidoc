85 QStringList LibraryTreeImpl::getLibraryList(LibraryTreeNode parent) const
86 {
87  QStringList list;
88 
89  if(boost::num_vertices(mGraph) < 2){
90  return list;
91  }
92  const auto vertex = mGraph.vertex(parent.id());
93  const auto firstEndIt = boost::adjacenct_vertices(vertex, mGraph);
94  for(auto it = firstEndIt.first; it != firstEndIt.second; ++it){
95  list << boost::get(boost::vertex_bundle, mGraph)[*it].name;
96  }

101 QStringList LibraryTreeImpl::toFlatList() const
102 {

105  if(boost::num_vertices(mGraph) < 2){
106  return list;
107  }
108  const auto firstEndIt = boost::vertices(mGraph);
109  for(auto it = firstEndIt.first; it != firstEndIt.second; ++it){
110  list << boost::get(boost::vertex_bundle, mGraph)[*it].name;
111  }
112  list.removeDuplicates();
