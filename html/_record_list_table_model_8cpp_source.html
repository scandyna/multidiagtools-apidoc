48 QVariant RecordListTableModel::data(const QModelIndex & index, int role) const
49 {
50  if(!index.isValid()){
51  return QVariant();
52  }
53  if( (role != Qt::DisplayRole) && (role != Qt::EditRole) ){
54  return QVariant();
55  }
56  Q_ASSERT(index.row() >= 0);
57  Q_ASSERT(index.row() < mRecordList.rowCount());
58  Q_ASSERT(index.column() >= 0);
59  if(index.column() >= mRecordList.columnCount(index.row())){
60  return QVariant();
61  }
62  return mRecordList.data(index.row(), index.column());
63 }

73 void RecordListTableModel::updateColumnCount()
74 {
75  const auto cmp = [](const Record & a, const Record & b){
76  return (a.columnCount() < b.columnCount());
77  };
78  const auto it = std::max_element(mRecordList.cbegin(), mRecordList.cend(), cmp);
79  if(it == mRecordList.cend()){
80  mColumnCount = 0;
81  }else{
82  mColumnCount = it->columnCount();
83  }
