91 void Logger::stop()
92 {
93  // If thread was never started, nothing to do
94  if(!mThread.joinable()){
95  return;
96  }
97  // Wait until all errors are logged to separate thread backends
98  {
99  std::unique_lock<std::mutex> lock(mMutex);
100  if(!mAllErrorsLogged){
101  mCv.wait(lock, [=]{return mAllErrorsLogged;});
102  }
103  // Tell thread that it must stop
104  mEventForThread = End;
105  mCv.notify_one();
106  }
107  // Join thread
108  if(mThread.joinable()){
109  mThread.join();
110  }
111  // Wait until all errors are logged to main thread backends
112  QCoreApplication::processEvents(QEventLoop::AllEvents);
113 }

139 void Logger::run()
140 {
141  Error error;
142  bool running = true;
143 
144  // Work..
145  while(running){
146  /*
147  * Wait for new job or end:
148  * We must lock the mutex before wait.
149  * wait will then relese it.
150  * Once wait returns, the mutex is locked again.
151  * We only want to lock it to take a error from queue,
152  * not all the time.
153  */
154  {
155  std::unique_lock<std::mutex> lock(mMutex);
156  mCv.wait(lock, [=]{return mEventForThread != NoEvent;});
157  }
158  // Log all available errors
159  do{
160  error = takeError();
161  if(!error.isNull()){
162  outputErrorToSeparateThreadBackends(error);
163  }
164  }while(!error.isNull());
165  // Notify that we logged all available errors
166  {
167  std::unique_lock<std::mutex> lock(mMutex);
168  // Check special case of stopping
169  if(mEventForThread == End){
170  running = false;
171  }else{
172  mEventForThread = NoEvent;
173  }
174  mAllErrorsLogged = true;
175  }
176  mCv.notify_one();
177  }
178 }
