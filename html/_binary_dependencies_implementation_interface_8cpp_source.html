104 bool BinaryDependenciesImplementationInterface::listContainsLibrary(const QStringList& list, const LibraryInfo& li)
105 {
106  const auto pred = [li](const QString & filePath){
107  const auto libName = LibraryName( QFileInfo(filePath).fileName() );
108  return ( QString::compare( libName.name(), li.libraryName().name(), Qt::CaseSensitive ) == 0 );
109  };
110  return ( std::find_if(list.cbegin(), list.cend(), pred) != list.cend() );
111 }
