99 Qt::ItemFlags ForeignKeyProxyModel::flags(const QModelIndex& index) const
100 {
101  if(!index.isValid()){
102  return PkFkProxyModelBase::flags(index);
103  }
104  const auto editableEnabledFlags = mMap.getMostRestrictiveFlagsForColumn(index.column());
105  if(editableEnabledFlags.isForeignKeyEditable() && editableEnabledFlags.isForeignKeyItemsEnabled()){
106  return PkFkProxyModelBase::flags(index);
107  }
108  auto f = PkFkProxyModelBase::flags(index);
109  if(!editableEnabledFlags.isForeignKeyEditable()){
110  f &= Qt::ItemFlags(~Qt::ItemIsEditable);
111  }
112  if(!editableEnabledFlags.isForeignKeyItemsEnabled()){
113  f &= Qt::ItemFlags(~Qt::ItemIsEnabled);
114  }
115  return f;
116 }
