32 LibraryName::LibraryName(const QString & fullName)
33 {
34  if(fullName.isEmpty()){
35  return;
36  }
37  mFullName = fullName;
38  // Isolate the lib prefix
39  int start = 0;
40  if(fullName.startsWith(QLatin1String("lib"), Qt::CaseInsensitive)){
41  mPrefix = mFullName.left(3);
42  start = 3;
43  }
44  // Get extension and so version
45  QStringRef extension;
46  mVersion = parseExtension(mFullName, extension);
47  mExtension = extension.toString();
48  // Extract name
49  if(extension.isNull()){
50  mName = mFullName.mid(start);
51  }else{
52  const int extensionSize = fullName.size() - extension.position();
53  mName = mFullName.mid( start, fullName.size() - start - extensionSize - 1 );
54  }
55 }

94 LibraryVersion LibraryName::parseExtension(const QString& fullName, QStringRef& extension)
95 {
96  static const char *extensionRegEx = "(.so|.dll)";
97  const int extentionIndex = fullName.lastIndexOf( QRegularExpression(extensionRegEx, QRegularExpression::CaseInsensitiveOption) );
98  if(extentionIndex >= 0){
99  /*
100  * Extract extension
101  * We can have a versionned so name or simple extension
102  */
103  if(fullName.midRef(extentionIndex, 4) == QLatin1String(".so.")){
104  extension = fullName.midRef(extentionIndex+1, 2);
105  const int versionStartIndex = extentionIndex + 4;
106  const int versionLen = fullName.size() - versionStartIndex;
107  return LibraryVersion( fullName.midRef(versionStartIndex, versionLen) );
108  }
109  extension = fullName.midRef(extentionIndex+1);
110  }
111  return LibraryVersion();
112 }
