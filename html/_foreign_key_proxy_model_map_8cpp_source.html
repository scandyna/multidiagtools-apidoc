108 ForeignKeyProxyModelMapItemFlags ForeignKeyProxyModelMap::getMostRestrictiveFlagsForColumn(int column) const
109 {
110  Q_ASSERT(column >= 0);
111 
112  ForeignKeyProxyModelMapItemFlags flags;
113 
114  for(const auto & item : mMap){
115  if(item.foreignKeyContainsColumn(column)){
116  if(!item.isForeignKeyEditable()){
117  flags.setForeignKeyEditable(false);
118  }
119  if(!item.isForeignKeyItemsEnabled()){
120  flags.setForeignKeyItemsEnabled(false);
121  }
122  }
123  }
124
