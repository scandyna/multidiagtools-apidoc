39 void ConsoleBackend::logError(const Error & error)
40 {
41  QDebug dbg(QtDebugMsg);
42  QDebugStateSaver stateSaver(dbg);
43  QString msg;
44 
45  // Setup debugger
46  dbg.resetFormat();
47  dbg.noquote();
48  dbg.nospace();
49  /*
50  * We build the string and then print it out
51  * (prevents splitting of informations of the same error)
52  * Note: because potentially all errors are logged,
53  * we not have to print the error stack here.
54  */
55  msg = getErrorStackString(error) + getErrorString(error);
56  dbg << msg;
57 }

59 QString ConsoleBackend::getErrorString(const Error & error) const
60 {
61  QString str;
62  QString informativeText;
63 
64  str = tr("Error occured in ") % error.functionName() % QStringLiteral("\n") \
65  % QStringLiteral("-> ") % tr("Text: ") % error.text() % QStringLiteral("\n");
66  informativeText = error.informativeText();
67  if(!informativeText.isEmpty()){
68  str += QStringLiteral("-> ") % tr("Informative text: ") % error.informativeText() % QStringLiteral("\n");
69  }
70  str += QStringLiteral("-> ") % tr("File: ") % error.fileName() + QStringLiteral("\n") \
71  % QStringLiteral("-> ") % tr("Line: ") % QString::number(error.fileLine());
72 
73  return str;
74 }
