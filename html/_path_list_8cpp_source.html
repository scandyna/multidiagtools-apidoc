80 PathList PathList::getSystemExecutablePathList()
81 {
82  PathList pathList;
83 
84  const auto pathEnv = QString::fromLocal8Bit( qgetenv("PATH") );
85 #ifdef Q_OS_WIN
86  const auto separator = QChar::fromLatin1(';');
87 #else
88  const auto separator = QChar::fromLatin1(':');
89 #endif
90  const auto rawPathList = pathEnv.split(separator);
91  for(const auto & path : rawPathList){
92  pathList.appendPath( QDir::cleanPath(path) );
93  }
94 
95  return pathList;
96 }
