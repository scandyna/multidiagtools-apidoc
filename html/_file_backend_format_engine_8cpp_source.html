30 QString FileBackendFormatEngine::formatError(const Error & error) const
31 {
33  QString str;
34 
35  // Prepare line beginning
36  const QString lineBegin = QDateTime::currentDateTime().toString(QLatin1String("yyyy.MM.dd hh:mm:ss")) % QLatin1Char(' ') \
37  % errorLevelText(error.level()) % QLatin1String(": ");
38  // Build message
39  str = lineBegin % QLatin1String("In ") % error.functionName() % QLatin1Char('\n') \
40  % lineBegin + QLatin1String("-> ") % tr("Text: ") % error.text() % QLatin1Char('\n');
41  const auto informativeText = error.informativeText();
42  if(!informativeText.isEmpty()){
43  str += lineBegin % QLatin1String("-> ") % error.informativeText() % QLatin1Char('\n');
44  }
45  str += lineBegin % QLatin1String("-> ") % tr("File: ") % error.fileName() % QLatin1Char('\n') \
46  % lineBegin % QLatin1String("-> ") % tr("Line: ") % QString::number(error.fileLine());
47 
48  return str;
49 }
