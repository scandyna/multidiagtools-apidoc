87 bool RPath::prependPathForBinaries(const QString & path, const QString& directoryPath)
88 {
89  QDir dir(directoryPath);
90 
91  if(!dir.exists()){
92  const QString msg = tr("Directory '%1' does not exist, or is a file.").arg(directoryPath);
93  auto error = mdtErrorNewQ(msg, Mdt::Error::Critical, this);
94  setLastError(error);
95  return false;
96  }
97  const auto fileInfoList = dir.entryInfoList(QDir::Files);
98  for(const auto fileInfo : fileInfoList){
99  if(BinaryFormat::isFileAnExecutableByExtension(fileInfo)){
100  Console::info(2) << " prepend path '" << path << "' to '" << fileInfo.fileName();
101  if(!prependPath(path, fileInfo.absoluteFilePath())){
102  return false;
103  }
104  }
105  }
106 
107  return true;
108 }
