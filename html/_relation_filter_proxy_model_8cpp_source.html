166 void RelationFilterProxyModel::onParentModelDataChanged(const QModelIndex& topLeft, const QModelIndex& bottomRight, const QVector<int> & roles)
167 {
168  if( (!roles.isEmpty()) && (!roles.contains(Qt::EditRole)) && (!roles.contains(Qt::DisplayRole)) ){
169  return;
170  }
171  /*
172  * We only act on currently filtered rows.
173  * Those are filtered regarding parent model row
174  */
175  RowRange r;
176  r.setFirstIndex(topLeft);
177  r.setLastIndex(bottomRight);
178  if(!r.containsRow(mParentModelRow)){
179  return;
180  }
181  ColumnRange c;
182  c.setFirstIndex(topLeft);
183  c.setLastIndex(bottomRight);
184  mKeyCopier->copyKeyData( getCurrentSourceModelRowList(), c );
185  invalidateFilter();
186 }

188 bool RelationFilterProxyModel::filterAcceptsRow(int source_row, const QModelIndex & source_parent) const
189 {
190  if(mParentModelRow < 0){
191  return false;
192  }
193  if(source_parent.isValid()){
194  return false;
195  }
196  Q_ASSERT(!mParentModel.isNull());
197  Q_ASSERT(sourceModel() != nullptr);
198  Q_ASSERT(mParentModelRow >= 0);
199  Q_ASSERT(mParentModelRow < mParentModel->rowCount());
200 // if(mParentModel.isNull() || (sourceModel() == nullptr)){
201 // return false;
202 // }
203 // if( (mParentModelRow < 0) || (mParentModelRow >= mParentModel->rowCount()) ){
204 // return false;
205 // }
206  if(mFilterExpression.isNull()){
207  return true;
208  }
209  /*
210  * After a model reset, possibly due to changing source model,
211  * filter expression possibly does not match until setFilter() is called again.
212  * We don't want to constraint setup order, so do bound checking here.
213  */
214  if( sourceModel()->columnCount() < (mFilterExpression.greatestColumn()+1) ){
215  return false;
216  }
217  return mFilterExpression.eval(sourceModel(), source_row, ParentModelEvalData(mParentModel, mParentModelRow), filterCaseSensitivity());
218 }
