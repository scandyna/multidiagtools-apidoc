56 bool CompareLikeTo::isLike(const FilterColumnData & col, const QString & like, const FilterEvalData & data) const
57 {
58  using Mdt::FilterExpression::LikeExpressionRegexTransform;
59 
60  QRegularExpression regex;
61 
62  if(data.caseSensitivity() == Qt::CaseInsensitive){
63  regex.setPatternOptions(QRegularExpression::CaseInsensitiveOption);
64  }
65  regex.setPattern( LikeExpressionRegexTransform::getRegexPattern(like) );
66  Q_ASSERT(regex.isValid());
67 
68  return regex.match( getStringValue(col, data) ).hasMatch();
