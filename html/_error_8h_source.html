369  template <typename T>
370  T error() const
371  {
372  static_assert(!std::is_reference<T>::value, "Getting a reference type of stored error is not possible");
373 
374  if(!pvShared){
375  return T();
376  }
377  #ifndef QT_NO_DEBUG
378  /*
379  * Requested type coherence check:
380  * If we can, we provide caller information.
381  * This seems somwhat ugly, but provides helpfull informations
382  * (and is only compiled in debug)
383  * Note: we use Q_ASSERT_X() because qFatal() requires to include QDebug,
384  * which we really not want here.
385  */
386  QString msg;
387  if(std::type_index(typeid(T)) != pvShared->userErrorType){
388  // If we have caller informations, put it into failure message
389  const QString functionName = pvShared->functionName;
390  QString caller;
391  if(!functionName.isEmpty()){
392  caller = QString(QLatin1String(" Error object source: %1 in file %2, line %3"))
393  .arg(functionName, pvShared->fileName)
394  .arg(pvShared->lineNumber);
395  }
396  msg = QLatin1String("Requested type T is not the same as stored error type.") + caller;
397  }
398  Q_ASSERT_X(std::type_index(typeid(T)) == pvShared->userErrorType, "T Error::error() const", msg.toStdString().c_str());
399  Q_ASSERT(dynamic_cast<const ErrorPrivate<T>*>(pvShared.constData()) != nullptr);
400  #endif // #ifndef QT_NO_DEBUG
401 
402  return static_cast<const ErrorPrivate<T>*>(pvShared.constData())->error;
403  }
