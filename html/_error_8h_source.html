368  template <typename T>
369  T error() const
370  {
371  static_assert(!std::is_reference<T>::value, "Getting a reference type of stored error is not possible");
372 
373  if(!pvShared){
374  return T();
375  }
376  #ifndef QT_NO_DEBUG
377  /*
378  * Requested type coherence check:
379  * If we can, we provide caller information.
380  * This seems somwhat ugly, but provides helpfull informations
381  * (and is only compiled in debug)
382  * Note: we use Q_ASSERT_X() because qFatal() requires to include QDebug,
383  * which we really not want here.
384  */
385  QString msg;
386  if(std::type_index(typeid(T)) != pvShared->userErrorType){
387  // If we have caller informations, put it into failure message
388  const QString functionName = pvShared->functionName;
389  QString caller;
390  if(!functionName.isEmpty()){
391  caller = QString(QLatin1String(" Error object source: %1 in file %2, line %3"))
392  .arg(functionName, pvShared->fileName)
393  .arg(pvShared->lineNumber);
394  }
395  msg = QLatin1String("Requested type T is not the same as stored error type.") + caller;
396  }
397  Q_ASSERT_X(std::type_index(typeid(T)) == pvShared->userErrorType, "T Error::error() const", msg.toStdString().c_str());
398  Q_ASSERT(dynamic_cast<const ErrorPrivate<T>*>(pvShared.constData()) != nullptr);
399  #endif // #ifndef QT_NO_DEBUG
400 
401  return static_cast<const ErrorPrivate<T>*>(pvShared.constData())->error;
402  }
