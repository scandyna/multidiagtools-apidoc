47 bool CsvFileParser::openFile(const QFileInfo & fileInfo, const QByteArray & encoding)
48 {
49  // Close possibly previously open file
50  if(mFile.isOpen()){
51  closeFile();
52  }
53  // Open file
54  mFile.setFileName(fileInfo.absoluteFilePath());
55  if(!mFile.open(QIODevice::ReadOnly)){
56  const auto msg = tr("Could not open file '%1'\nDirectory: '%2'").arg(fileInfo.fileName(), fileInfo.dir().absolutePath());
57  mLastError = mdtErrorNew(msg, Mdt::Error::Critical, "CsvFileParser");
58  mLastError.stackError(mdtErrorFromQFile(mFile, "CsvFileParser"));
59  mLastError.commit();
60  return false;
61  }
62  // Assign file iterator
63  if(!mFileIterator.setSource(&mFile, encoding)){
64  mLastError = mFileIterator.lastError();
65  return false;
66  }
67 
68  return true;
69 }

87 Expected<StringRecord> CsvFileParser::readLine()
88 {
89  Q_ASSERT_X(mParser->isValid(), "CsvFileParser", "No CSV settings set");
90 
91  Expected<StringRecord> record;
92 
93  // Create multi pass iterators
94  auto first = makeFileMultiPassIterator(mFileIterator);
95  auto last = makeFileMultiPassIterator(FileInputIterator());
96  // Parse a line
97  record = mParser->readLine(first, last);
98  if(!record){
99  // Maybe a error on file
100  if(mFileIterator.errorOccured()){
101  record.error().stackError(mFileIterator.lastError());
102  }
103  }
104 
105  return record;
106 }
