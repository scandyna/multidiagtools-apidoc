193 Mdt::Expected<TranslationInfoList> findTranslationsInDirectory(const QStringList& qmFileNames, const QString& directoryPath)
194 {
195  TranslationInfoList translations;
196 
197  QDir dir(directoryPath);
198  if(!dir.exists()){
199  const auto msg = QCoreApplication::translate("findTranslationsInDirectory()","Directory '%1' does not exist.").arg(directoryPath);
200  auto error = mdtErrorNew(msg, Mdt::Error::Critical, "");
201  return error;
202  }
203  const auto fiList = dir.entryInfoList(QDir::Files);
204  for(const auto & fi : fiList){
205  if(qmFileNames.contains(fi.fileName(), Qt::CaseInsensitive)){
206  translations.addTranslation( TranslationInfo::fromQmFilePath(fi.absoluteFilePath()) );
207  }
208  }
209 
210  return translations;
211 }
