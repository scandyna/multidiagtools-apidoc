85 void Error::stackError(const Error & error)
86 {
87  Q_ASSERT(pvShared);
88  Q_ASSERT(error.pvShared);
89 
90  pvShared.detach();
91  /*
92  * We push errors back (as if it was a queue),
93  * Then, when getting them back, we do is reverse order
94  */
95  // Copy given error's stack if available
96  auto first = error.pvShared->pvErrorStack.cbegin();
97  auto last = error.pvShared->pvErrorStack.cend();
98  std::copy(first, last, std::back_inserter(pvShared->pvErrorStack));
99  // Push error itself at end
100  pvShared->pvErrorStack.push_back(error);
101 }

103 std::vector<Error> Error::getErrorStack() const
104 {
105  std::vector<Error> stack;
106 
107  if(!pvShared){
108  return stack;
109  }
110  if(pvShared->pvErrorStack.empty()){
111  return stack;
112  }
113  auto first = pvShared->pvErrorStack.cbegin();
114  auto last = pvShared->pvErrorStack.cend();
115  std::reverse_copy(first, last, std::back_inserter(stack));
116 
117  return stack;
118 }

172 Error::Level _mdtErrorlevelFromQFileDeviceFileError(QFileDevice::FileError fileError)
173 {
174  switch(fileError){
175  case QFileDevice::NoError:
176  return Error::NoError;
177  case QFileDevice::ReadError:
178  case QFileDevice::WriteError:
179  case QFileDevice::FatalError:
180  case QFileDevice::ResourceError:
181  case QFileDevice::OpenError:
182  case QFileDevice::AbortError:
183  case QFileDevice::TimeOutError:
184  case QFileDevice::UnspecifiedError:
185  case QFileDevice::RemoveError:
186  case QFileDevice::RenameError:
187  case QFileDevice::PositionError:
188  case QFileDevice::ResizeError:
189  case QFileDevice::PermissionsError:
190  case QFileDevice::CopyError:
191  return Error::Critical;
192  }
193  return Error::Critical;
194 }
