107 bool BinaryDependenciesObjdump::findAndAddDependenciesForNode(const QString& binaryFilePath, LibraryTreeNode node)
108 {
109  /*
110  * If a library allready exists in the tree,
111  * we just add it as child of the node.
112  * Else, we search the dependencies.
113  */
114  if(mAlreadyProcessedLibraries.contains(binaryFilePath)){
115  mLibraryTree.addLibrary(binaryFilePath, node);
116  return true;
117  }
118  Console::info(3) << " processing " << QFileInfo(binaryFilePath).fileName();
119 
120  ObjdumpWrapper objdump;
121  if(!objdump.execFindDependencies(binaryFilePath)){
122  setLastError(objdump.lastError());
123  return false;
124  }
125 
126  ObjdumpDependenciesParser parser;
127  if(!parser.parse( objdump.readAllStandardOutputString() )){
128  const QString msg = tr("Parsing objdump output for file '%1' failed.").arg(binaryFilePath);
129  auto error = mdtErrorNewQ(msg, Mdt::Error::Critical, this);
130  setLastError(error);
131  return false;
132  }
133 
134  const auto rawDependenciesRecordList = parser.rawDependencies();
135  for(const auto & record : rawDependenciesRecordList){
136  Q_ASSERT(record.columnCount() == 1);
137  if(!isLibraryInExcludeList(record)){
138  Library library;
139  if( !library.findLibrary(record.data(0), mLibrarySearchPathList, Library::ExcludeSystemPaths) ){
140  const QString msg = tr("Could not find library '%1'.\nSearched in %2")
141  .arg(record.data(0)).arg(mLibrarySearchPathList.toStringList().join(", "));
142  auto error = mdtErrorNewQ(msg, Mdt::Error::Critical, this);
143  setLastError(error);
144  return false;
145  }
146  mAlreadyProcessedLibraries.append(binaryFilePath);
147  const auto libraryFilePath = library.libraryInfo().absoluteFilePath();
148  const auto libraryNode = mLibraryTree.addLibrary( libraryFilePath, node );
149  if(!findAndAddDependenciesForNode( libraryFilePath, libraryNode )){
150  return false;
151  }
152  }
153  }
154 
155  return true;
156 }

158 bool BinaryDependenciesObjdump::isLibraryInExcludeList(const StringRecord & record)
159 {
160  Q_ASSERT(record.columnCount() > 0);
161 
162  const auto cmp = [record](const char * const excludeName){
163  LibraryName libName(record.data(0));
164  return ( QString::compare( libName.name(), QLatin1String(excludeName), Qt::CaseInsensitive ) == 0);
165  };
166  const auto it = std::find_if( LibrayExcludeListWindows.cbegin(), LibrayExcludeListWindows.cend(), cmp );
167  return (it != LibrayExcludeListWindows.cend());
168 }
