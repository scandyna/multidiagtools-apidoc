58 bool BinaryDependenciesLdd::findDependencies(const QString& binaryFilePath)
59 {
60  Console::info(2) << " searching dependencies for " << binaryFilePath;
61 
62  LddWrapper ldd;
63  if(!ldd.execFindDependencies(binaryFilePath)){
64  setLastError(ldd.lastError());
65  return false;
66  }
67 
68  LddDependenciesParser parser;
69  if( !parser.parse( ldd.readAllStandardOutputString() ) ){
70  const QString msg = tr("Parsing ldd output for file '%1' failed.").arg(binaryFilePath);
71  auto error = mdtErrorNewQ(msg, Mdt::Error::Critical, this);
72  setLastError(error);
73  return false;
74  }
75 
76  auto list = parser.rawDependencies();
77  fillAndSetDependencies(list);
78  if(!mNotFoundDependencies.isEmpty()){
79  const QString msg = tr("Some dependencies have not been found for file '%1': %2")
80  .arg( binaryFilePath, stringRecordListToStringNameList(mNotFoundDependencies).join(", ") );
81  auto error = mdtErrorNewQ(msg, Mdt::Error::Critical, this);
82  setLastError(error);
83  return false;
84  }
85 
86  if(Console::level() >= 3){
87  Console::info(3) << " found dependencies:";
88  for(const auto & library : dependencies() ){
89  Console::info(3) << " " << library.libraryName().fullName();
90  }
91  }
92 
93  return true;
94 }

96 void BinaryDependenciesLdd::fillAndSetDependencies(PlainText::StringRecordList& data)
97 {
98  PlainText::StringRecordList dependencies;
99 
100  setLibrariesNameIfMissing(data);
101  // Move libraries that are not found to <notFoundDependencies> list
102  mNotFoundDependencies.clear();
103  Mdt::Algorithm::moveIf( data, mNotFoundDependencies, isLibraryNotFound );
104  // Move libraries that are not in the exclude list to <dependencies>
105  Mdt::Algorithm::moveIf( data, dependencies, isLibraryNotInExcludeList );
106  // Build and set the resulting dependencies
107  setDependencies( stringRecordListToLibraryInfoList(dependencies) );
108 }

147 bool BinaryDependenciesLdd::isLibraryNotInExcludeList(const PlainText::StringRecord & record)
148 {
149  Q_ASSERT(record.columnCount() > 0);
150 
151  const auto cmp = [record](const char * const excludeName){
152  LibraryName libName(record.data(0));
153  return ( QString::compare( libName.name(), QLatin1String(excludeName), Qt::CaseSensitive ) == 0);
154  };
155  const auto it = std::find_if( LibrayExcludeListLinux.cbegin(), LibrayExcludeListLinux.cend(), cmp );
156  return (it == LibrayExcludeListLinux.cend());
157 }
