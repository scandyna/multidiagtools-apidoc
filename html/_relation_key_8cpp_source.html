37 void RelationKey::setKey(const PrimaryKey & parentModelPk, const ForeignKey & childModelFk)
38 {
39  Q_ASSERT(!parentModelPk.isNull());
40  Q_ASSERT(!childModelFk.isNull());
41  Q_ASSERT(parentModelPk.columnCount() == childModelFk.columnCount());
42 
43  clear();
44  mColumnPairList.reserve(parentModelPk.columnCount());
45  const auto op = [](int parentModelColumn, int childModelColumn){
46  return RelationColumnPair(parentModelColumn, childModelColumn);
47  };
48  std::transform(parentModelPk.begin(), parentModelPk.end(), childModelFk.begin(), std::back_inserter(mColumnPairList), op);
49 }

51 bool RelationKey::containsColumnPair(int parentModelColumn, int childModelColumn) const
52 {
53  const auto pred = [parentModelColumn, childModelColumn](const RelationColumnPair & pair){
54  return ( pair.parentModelColumn() == parentModelColumn && pair.childModelColumn() == childModelColumn );
55  };
56  return ( std::find_if(mColumnPairList.cbegin(), mColumnPairList.cend(), pred) != mColumnPairList.cend() );
57 }

59 int RelationKey::greatestParentModelColumn() const
60 {
61  const auto cmp = [](const RelationColumnPair & a, const RelationColumnPair & b){
62  return (a.parentModelColumn() < b.parentModelColumn());
63  };
64  const auto it = std::max_element(mColumnPairList.cbegin(), mColumnPairList.cend(), cmp);
65  if(it == mColumnPairList.cend()){
66  return -1;
67  }
68  return it->parentModelColumn();
69 }
