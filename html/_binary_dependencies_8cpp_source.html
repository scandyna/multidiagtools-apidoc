105 std::unique_ptr<BinaryDependenciesImplementationInterface> BinaryDependencies::initImpl(const QFileInfo & binaryFileInfo, const PathList & searchFirstPathPrefixList)
106 {
107  std::unique_ptr<BinaryDependenciesImplementationInterface> impl;
108  SearchPathList searchPathList;
109 
110  // Check that given file exists
111  if(!binaryFileInfo.exists()){
112  const auto msg = tr("File '%1' does not exist.").arg(binaryFileInfo.absoluteFilePath());
113  auto error = mdtErrorNewQ(msg, Mdt::Error::Critical, this);
114  setLastError(error);
115  return impl;
116  }
117  // Choose a implementation regarding binary file format
118  BinaryFormat bfmt;
119  if(!bfmt.readFormat(binaryFileInfo.absoluteFilePath())){
120  setLastError(bfmt.lastError());
121  return impl;
122  }
123  switch(bfmt.operatindSystem()){
124  case OperatingSystem::Unknown:
125  break;
126  case OperatingSystem::Linux:
127  impl.reset(new BinaryDependenciesLdd);
128  break;
129  case OperatingSystem::Windows:
130  impl.reset(new BinaryDependenciesObjdump);
131  searchPathList.prependPath( binaryFileInfo.absoluteDir().absolutePath() );
132  searchPathList.setPathPrefixList(searchFirstPathPrefixList);
133  searchPathList.setPathSuffixList({"bin","qt5/bin"});
134  impl->setLibrarySearchFirstPathList(searchPathList.pathList());
135  break;
136  }
137  if(!impl){
138  const QString msg = tr("Could not find a tool to get dependencies for file '%1'.").arg(binaryFileInfo.absoluteFilePath());
139  auto error = mdtErrorNewQ(msg, Mdt::Error::Critical, this);
140  setLastError(error);
141  return impl;
142  }
143 
144  return impl;
145 }
