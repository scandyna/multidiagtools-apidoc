32 QmFileName::QmFileName(const QString & fullName)
33 {
34  mFullName = fullName;
35  const auto fullBaseName = QFileInfo(fullName).baseName();
36  const auto items = fullBaseName.split(QLatin1Char('_'), QString::SkipEmptyParts);
37 
38  if(items.isEmpty()){
39  return;
40  }
41  if(items.size() == 1){
42  // We have only base name
43  mBaseName = fullBaseName;
44  }else if(items.size() == 2){
45  // We have maybe base name and language
46  if( isLanguageCode(items.at(1)) ){
47  mBaseName = items.at(0);
48  mLanguageCode = LanguageCode::fromString( items.at(1) );
49  }else{
50  mBaseName = fullBaseName;
51  }
52  }else{
53  Q_ASSERT(items.size() >= 3);
54  /*
55  * We can have:
56  * - base
57  * - base language
58  * - base language country
59  * Note: base can be some_base_name
60  */
61  const auto lastIndex = items.size() - 1;
62  if( isCountryCode(items.at(lastIndex)) && isLanguageCode(items.at(lastIndex-1)) ){
63  setBaseName(items, lastIndex-2);
64  mLanguageCode = LanguageCode::fromString( items.at(lastIndex-1) );
65  mCountrySuffix = items.at(lastIndex);
66  }else if( isLanguageCode(items.at(lastIndex)) ){
67  mLanguageCode = LanguageCode::fromString( items.at(lastIndex) );
68  setBaseName(items, lastIndex-1);
69  }else{
70  mBaseName = fullBaseName;
71  }
72  }
73 }
