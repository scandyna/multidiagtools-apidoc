78  void setupParser(const CsvParserSettings & settings)
79  {
80  Q_ASSERT(settings.isValid());
81 
82  namespace qi = boost::spirit::qi;
83  namespace phoenix = boost::phoenix;
84  // Parsers
85  using qi::lit;
86  using qi::eol;
87  using boost::spirit::standard_wide::char_;
88  using boost::spirit::standard_wide::space;
89 
90  const char fieldSep = settings.fieldSeparator();
91  const char fieldQuote = settings.fieldProtection();
92  const bool parseExp = settings.parseExp();
93 
94  /*
95  * Give our rules a name (used for error)
96  */
97  mRecordRule.name("RecordRule");
98  /*
99  * Build the grammar
100  * This grammar is based on CSV-1203.
101  * Some rules are also altered, to try to match RFC 4180 and unicode
102  */
103  mRecordRule = mRecordPayload >> -eol; // RFC 4180 do not need a end of line in last line
104  mRecordPayload = mFieldColumn % char_(fieldSep);
105  mFieldColumn = mProtectedField | mUnprotectedField;
106  if(parseExp){
107  mProtectedField = lit(fieldQuote) >> -lit('~') >> mFieldPayload >> lit(fieldQuote);
108  mUnprotectedField = -lit('~') >> mRawFieldPayload;
109  }else{
110  mProtectedField = lit(fieldQuote) >> mFieldPayload >> lit(fieldQuote);
111  mUnprotectedField = -char_('~') >> mRawFieldPayload; // mUnprotectedField = mRawFieldPayload causes runtime exception
112  }
113  mFieldPayload = *mAnychar;
114  mRawFieldPayload = *mSafechar | (mSafechar >> *char_ >> mSafechar);
115  // Character collections
116  mAnychar = mChar | char_(fieldSep) | (char_(fieldQuote) >> lit(fieldQuote)) | space; // space matches space, CR, LF and other See std::isspace()
117  mChar = mSafechar | char_(0x20); // 0x20 == SPACE char
119  std::string exclude = std::string("\n\t\r") + fieldSep + fieldQuote;
120  mSafechar = ~char_(exclude);
121 
122  /*
123  * Parser nodes debug.
124  * Don't forget to also uncomment #define BOOST_SPIRIT_DEBUG
125  * Note: to display what id parsed,
126  * a certain amount of data is read from source.
127  * With mdtCsvFileParser, this causes
128  * it to be at eof after the first call of readLine().
129  */

155  Mdt::Expected<StringRecord> readLine(SourceIterator & first, const SourceIterator & last)
156  {
157  Q_ASSERT(first != last);
158  Q_ASSERT_X(isValid(), "CsvParserTemplate", "parser setup never done");
159 
160  StringRecord record;
161 
162  bool ok = boost::spirit::qi::parse(first, last, mRecordRule, record);
163  if(!ok){
164  first = last;
166  const auto msg = QCoreApplication::translate("Mdt::PlainText::CsvParserTemplate", "Parsing error occured.");
167  auto error = mdtErrorNew(msg, Mdt::Error::Critical, "CsvParserTemplate");
168  return error;
170  }
171 
172  return record;
173  }
